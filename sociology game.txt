/* ---------- engine below: no need to edit ---------- */
let dragged = null;
let currentCount = 0;

function shuffle(a){for(let i=a.length-1;i>0;i--){const j=Math.floor(Math.random()*(i+1));[a[i],a[j]]=[a[j],a[i]];}return a;}

function makeTile(item,idx){
  const t=document.createElement('div');
  t.className='tile';
  t.draggable=true;
  t.dataset.answer=item.answer;
  t.dataset.idx=idx;
  t.innerHTML=`<div class="tile-text">${item.text}</div><div class="reveal-note">${item.note}</div>`;
  t.addEventListener('dragstart',e=>{dragged=t;t.classList.add('dragging');e.dataTransfer.effectAllowed='move';});
  t.addEventListener('dragend',()=>{t.classList.remove('dragging');});
  // touch support
  t.addEventListener('touchstart',touchStart,{passive:false});
  return t;
}

function pickSubset(){
  if(!DRAW||DRAW>=ITEMS.length) return ITEMS.map((it,i)=>({it,i}));
  const indexed=ITEMS.map((it,i)=>({it,i}));
  const lefts=shuffle(indexed.filter(x=>x.it.answer==='left'));
  const rights=shuffle(indexed.filter(x=>x.it.answer==='right'));
  const half=Math.floor(DRAW/2);
  const chosen=lefts.slice(0,half).concat(rights.slice(0,DRAW-half));
  return shuffle(chosen);
}

function build(){
  const bank=document.getElementById('bank');
  bank.querySelectorAll('.tile').forEach(n=>n.remove());
  document.getElementById('left-tiles').innerHTML='';
  document.getElementById('right-tiles').innerHTML='';
  document.body.classList.remove('revealed');
  document.getElementById('score').textContent='';
  currentCount=0;
  pickSubset().forEach(({it,i})=>{bank.appendChild(makeTile(it,i));currentCount++;});
}

function attachZone(zoneEl){
  zoneEl.addEventListener('dragover',e=>{e.preventDefault();zoneEl.classList.add('over');});
  zoneEl.addEventListener('dragleave',()=>zoneEl.classList.remove('over'));
  zoneEl.addEventListener('drop',e=>{
    e.preventDefault();zoneEl.classList.remove('over');
    if(!dragged)return;
    const target=zoneEl.querySelector('.tiles')||zoneEl;
    target.appendChild(dragged);
    dragged.classList.add('placed');
  });
}

/* ---- touch drag (mobile/tablet) ---- */
let touchEl=null,ghost=null;
function touchStart(e){
  touchEl=e.currentTarget;
  const tch=e.touches[0];
  ghost=touchEl.cloneNode(true);
  ghost.style.position='fixed';ghost.style.pointerEvents='none';
  ghost.style.opacity='0.85';ghost.style.zIndex='999';ghost.style.width=touchEl.offsetWidth+'px';
  document.body.appendChild(ghost);
  moveGhost(tch);
  touchEl.classList.add('dragging');
  document.addEventListener('touchmove',touchMove,{passive:false});
  document.addEventListener('touchend',touchEnd);
  e.preventDefault();
}
function moveGhost(tch){if(ghost){ghost.style.left=(tch.clientX-ghost.offsetWidth/2)+'px';ghost.style.top=(tch.clientY-22)+'px';}}
function touchMove(e){const tch=e.touches[0];moveGhost(tch);e.preventDefault();}
function touchEnd(e){
  const tch=e.changedTouches[0];
  const el=document.elementFromPoint(tch.clientX,tch.clientY);
  const zone=el&&el.closest('.zone');
  const toBank=el&&el.closest('.bank');
  if(zone){(zone.querySelector('.tiles')||zone).appendChild(touchEl);touchEl.classList.add('placed');}
  else if(toBank){document.getElementById('bank').appendChild(touchEl);touchEl.classList.remove('placed');}
  if(ghost){ghost.remove();ghost=null;}
  if(touchEl)touchEl.classList.remove('dragging');
  touchEl=null;
  document.removeEventListener('touchmove',touchMove);
  document.removeEventListener('touchend',touchEnd);
}
